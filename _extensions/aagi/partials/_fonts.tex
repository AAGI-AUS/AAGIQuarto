% -------------------------
% Engine detection + fonts
% -------------------------
\usepackage{iftex}

\ifPDFTeX
  \usepackage{lmodern}
  \renewcommand{\familydefault}{\sfdefault}
\else
  \usepackage{fontspec}
  \usepackage{microtype}

  % Prefer Proxima Nova, fall back to Arial, then TeX Gyre Heros
  \IfFontExistsTF{Proxima Nova}{
    \setsansfont[Scale=MatchLowercase]{Proxima Nova}
  }{
    \IfFontExistsTF{Arial}{
      \setsansfont[Scale=MatchLowercase]{Arial}
    }{
      \setsansfont[Scale=MatchLowercase]{TeX Gyre Heros}
    }
  }

  \renewcommand{\familydefault}{\sfdefault}

  \setmonofont[
    Path        = assets/fonts/,
    Extension   = .ttf,
    UprightFont = FiraCode-Regular,
    BoldFont    = FiraCode-Bold,
    ItalicFont  = FiraCode-Light,
    Scale       = MatchLowercase
  ]{FiraCode-Regular}
\fi
