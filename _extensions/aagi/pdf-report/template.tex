%
% fontspec lets XeLaTeX resolve fonts by name from the system font catalogue.
% The chain below tries each font in order and uses the first one found:
%   1. Proxima Nova     — AAGI brand font, available on developer machines
%   2. Arial            — Microsoft/Apple system font, widely available
%   3. TeX Gyre Heros   — metric-compatible Helvetica clone, always present
%                         in a full TeX Live installation including CI runners
%
% TeX Gyre Heros is the guaranteed CI fallback. It is visually close to
% Helvetica/Arial and far preferable to Latin Modern Sans for AAGI branding.
%
% microtype works with XeLaTeX but only in limited mode; it is kept for
% the margin kerning and font expansion it does provide.
\usepackage{fontspec}
\usepackage{microtype}

% Sans-serif body font: Proxima Nova → Arial → TeX Gyre Heros
\IfFontExistsTF{Proxima Nova}{%
  \setsansfont{Proxima Nova}[
    BoldFont     = {Proxima Nova Bold},
    ItalicFont   = {Proxima Nova Italic},
    BoldItalicFont = {Proxima Nova Bold Italic}]
}{%
  \IfFontExistsTF{Arial}{%
    \setsansfont{Arial}[
      BoldFont     = {Arial Bold},
      ItalicFont   = {Arial Italic},
      BoldItalicFont = {Arial Bold Italic}]
  }{%
    % TeX Gyre Heros: always present in full TeX Live; Helvetica-compatible
    \setsansfont{TeX Gyre Heros}[
      BoldFont     = {TeX Gyre Heros Bold},
      ItalicFont   = {TeX Gyre Heros Italic},
      BoldItalicFont = {TeX Gyre Heros Bold Italic}]
  }%
}

% Monospace font: Fira Code → Courier New → TeX Gyre Cursor
\IfFontExistsTF{Fira Code}{%
  \setmonofont{Fira Code}
}{%
  \IfFontExistsTF{Courier New}{%
    \setmonofont{Courier New}
  }{%
    \setmonofont{TeX Gyre Cursor}
  }%
}
